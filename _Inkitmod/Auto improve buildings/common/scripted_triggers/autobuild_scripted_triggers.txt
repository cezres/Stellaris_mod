	or = {
		and = {
			exists = owner
			owner = { not = { has_country_flag = auto_build_7_8 } }
			has_resource = {
				type = minerals
				amount > 0
			}
		}
		and = {
			exists = owner
			owner = { not = { has_country_flag = auto_build_7_9 } }		
			has_resource = {
				type = food
				amount > 0
			}
		}
		and = {
			exists = owner
			owner = { not = { has_country_flag = auto_build_7_10 } }
			has_resource = {
				type = energy
				amount > 0
			}
		}
		and = {
